<script>
  // Chequeo de sesión
  const usuarioLogueado = JSON.parse(localStorage.getItem('usuarioLogueado'));
  if (!usuarioLogueado) {
    alert("Debes iniciar sesión para acceder al panel");
    window.location.href = "login.html";
  } else {
    document.getElementById('userInfo').innerText = `Usuario: ${usuarioLogueado.nombre} (${usuarioLogueado.email})`;
  }

  // Opciones de etapa disponibles
  const etapas = [
    "Pendiente",
    "Visitando Apartamento Modelo",
    "Precalificando con el Banco",
    "En Proceso",
    "OK",
    "Finalizado"
  ];

  // Simulación de turnos almacenados en localStorage
  // Estructura ejemplo:
  // [{ticket: "T-0001", fechaHora: "10/06/2025 10:00", etapa: "Pendiente"}, ...]
  let turnos = JSON.parse(localStorage.getItem('turnos') || '[]');

  // Función para guardar turnos localStorage
  function guardarTurnos() {
    localStorage.setItem('turnos', JSON.stringify(turnos));
  }

  // Crear fila para cada turno
  function crearFilaTurno(turno, index) {
    const tr = document.createElement('tr');

    const tdTicket = document.createElement('td');
    tdTicket.textContent = turno.ticket;
    tr.appendChild(tdTicket);

    const tdFechaHora = document.createElement('td');
    tdFechaHora.textContent = turno.fechaHora;
    tr.appendChild(tdFechaHora);

    const tdEtapa = document.createElement('td');
    const selectEtapa = document.createElement('select');

    etapas.forEach(etapa => {
      const option = document.createElement('option');
      option.value = etapa;
      option.textContent = etapa;
      if (etapa === turno.etapa) option.selected = true;
      selectEtapa.appendChild(option);
    });

    selectEtapa.addEventListener('change', () => {
      turnos[index].etapa = selectEtapa.value;
      guardarTurnos();
      enviarNotificacionWhatsApp(index);
    });

    tdEtapa.appendChild(selectEtapa);
    tr.appendChild(tdEtapa);

    return tr;
  }

  // Renderizar tabla con turnos
  function renderizarTabla() {
    const tbody = document.querySelector('#tablaTurnos tbody');
    tbody.innerHTML = "";
    turnos.forEach((turno, i) => {
      tbody.appendChild(crearFilaTurno(turno, i));
    });
  }

  // Función para enviar mensaje WhatsApp al siguiente turno en la secuencia
  // con mensaje fijo: "¡Hola! es turno, por favor pasa con nuestro Oficial de Ventas Bisono. Gracias por preferirnos. Constructora Bisono."
  function enviarNotificacionWhatsApp(turnoActualIndex) {
    // Obtener índice siguiente turno que está en etapa "Pendiente"
    const siguienteTurnoIndex = turnos.findIndex((t, i) => i > turnoActualIndex && t.etapa === "Pendiente");
    if (siguienteTurnoIndex === -1) {
      console.log("No hay siguiente turno pendiente para notificar.");
      return;
    }

    // Para demo solo mostramos alerta y consola
    // Aquí debes integrar con tu backend o API WhatsApp para enviar el mensaje real
    const siguienteTurno = turnos[siguienteTurnoIndex];

    const mensaje = encodeURIComponent("¡Hola! es turno, por favor pasa con nuestro Oficial de Ventas Bisono. Gracias por preferirnos. Constructora Bisono.");
    // Suponiendo que tienes el número de teléfono guardado en turno, por ahora demo con link genérico
    const urlWhatsApp = `https://wa.me/?text=${mensaje}`;

    alert(`Se enviaría notificación WhatsApp al siguiente turno ${siguienteTurno.ticket}`);
    console.log("Enviar WhatsApp a siguiente turno:", siguienteTurno.ticket);
    window.open(urlWhatsApp, "_blank");
  }

  // Función para cerrar sesión
  function cerrarSesion() {
    localStorage.removeItem('usuarioLogueado');
    window.location.href = 'login.html';
  }

  // Inicialización de ejemplo con turnos si no existen
  if (turnos.length === 0) {
    turnos = [
      { ticket: "T-0001", fechaHora: "2025-06-10 10:00", etapa: "Pendiente" },
      { ticket: "T-0002", fechaHora: "2025-06-10 10:10", etapa: "Pendiente" },
      { ticket: "T-0003", fechaHora: "2025-06-10 10:20", etapa: "Pendiente" },
    ];
    guardarTurnos();
  }

  // Renderizar tabla al cargar la página
  renderizarTabla();

</script>
