    <h2 id="form-title">Iniciar Sesión</h2>
    <input type="text" id="username" placeholder="Usuario" />
    <input type="password" id="password" placeholder="Contraseña" />
    <button onclick="handleAuth()" id="auth-button">Entrar</button>
    <div class="toggle" onclick="toggleForm()">Crear cuenta nueva</div>
  </div>

  <script>
    let isLogin = true;

    function toggleForm() {
      isLogin = !isLogin;
      document.getElementById("form-title").innerText = isLogin ? "Iniciar Sesión" : "Crear Cuenta";
      document.getElementById("auth-button").innerText = isLogin ? "Entrar" : "Registrar";
      document.querySelector(".toggle").innerText = isLogin ? "Crear cuenta nueva" : "Volver a Iniciar Sesión";
    }

    function handleAuth() {
      const username = document.getElementById("username").value;
      const password = document.getElementById("password").value;
      if (!username || !password) {
        alert("Por favor completa todos los campos");
        return;
      }

      if (isLogin) {
        const users = JSON.parse(localStorage.getItem("empleados") || "[]");
        const found = users.find(u => u.username === username && u.password === password);
        if (found) {
          localStorage.setItem("usuarioActivo", JSON.stringify(found));
          location.href = "admin-panel.html";
        } else {
          alert("Credenciales incorrectas");
        }
      } else {
        let users = JSON.parse(localStorage.getItem("empleados") || "[]");
        if (users.some(u => u.username === username)) {
          alert("Ese usuario ya existe");
          return;
        }
        users.push({ username, password });
        localStorage.setItem("empleados", JSON.stringify(users));
        alert("Cuenta creada con éxito, ahora puedes iniciar sesión");
        toggleForm();
      }
    }
  </script>
