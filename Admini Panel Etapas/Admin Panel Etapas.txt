    <thead>
      <tr>

        <th>Fecha y Hora</th>

  <script>
    const etapas = [
      "Pendiente",
      "Visitando Apartamento Modelo",
      "Precalificando con el Banco",
      "En Proceso",
      "OK",
      "Finalizado"
    ];

    let turnos = JSON.parse(localStorage.getItem("turnos")) || [];

    function renderTurnos() {
      const tbody = document.querySelector("#tabla-turnos tbody");
      tbody.innerHTML = "";
      turnos.forEach((turno, index) => {
        const tr = document.createElement("tr");
        tr.innerHTML = `
          <td>${turno.numero}</td>
          <td>${turno.fecha}</td>
          <td>
            <select onchange="cambiarEtapa(${index}, this.value)">
              ${etapas.map(et => `<option value="${et}" ${et === turno.etapa ? "selected" : ""}>${et}</option>`).join("")}
            </select>
          </td>
        `;
        tbody.appendChild(tr);
      });
    }

    function cambiarEtapa(index, nuevaEtapa) {
      turnos[index].etapa = nuevaEtapa;
      localStorage.setItem("turnos", JSON.stringify(turnos));
      if (nuevaEtapa !== "Pendiente") {
        notificarSiguiente(index);
      }
    }

    function notificarSiguiente(indexActual) {
      const siguiente = turnos.find((t, idx) => idx > indexActual && t.etapa === "Pendiente");
      if (siguiente && siguiente.telefono) {
        const mensaje = encodeURIComponent("¡Hola! es turno, por favor pasa con nuestro Oficial de Ventas Bisono. Gracias por preferirnos. Constructora Bisono.");
        const url = `https://wa.me/${siguiente.telefono}?text=${mensaje}`;
        window.open(url, "_blank");
      }
    }

    renderTurnos();
  </script>
